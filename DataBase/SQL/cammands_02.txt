    players_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    team VARCHAR(50),
    role VARCHAR(50),
    run_scored INT CHECK (run_scored >= 0),
    wicket_taken INT CHECK (wicket_taken >= 0),
    auction_price INT,
    
);
